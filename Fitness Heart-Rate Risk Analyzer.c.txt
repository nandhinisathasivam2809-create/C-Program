#include <stdio.h>

int main() {
    int N, rate;
    int maxRate, dangerCount = 0;

    scanf("%d", &N);

    scanf("%d", &rate);
    maxRate = rate;

    if(rate > 140) {
        dangerCount++;
    }

    for(int i = 1; i < N; i++) {
        scanf("%d", &rate);

        if(rate > maxRate) {
            maxRate = rate;
        }

        if(rate > 140) {
            dangerCount++;
        }
    }

    printf("Max Heart Rate: %d\n", maxRate);
    printf("Danger Readings: %d\n", dangerCount);

    return 0;
}
